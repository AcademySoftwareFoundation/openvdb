vdb_tool 10.8.0

# This example demonstrates how to read a narrow-band level set from
# an OpenVDB file, convert it into an adaptive polygon mesh, and write
# the result to a binary PLY file.

# Read OpenVDB file with level set
read bunny.vdb

# Level set to polygon mesh conversion:
#   mesh adaptivity is 0.1 %
ls2mesh adapt=0.1

# Write polygon mesh to binary PLY file
write bunny.ply
