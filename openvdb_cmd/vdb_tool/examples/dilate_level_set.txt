vdb_tool 10.8.0

# This example demonstrates how to read a narrow-band level set from an
# OpenVDB file, dilate it, and write the result to a different OpenVDB file.

# Read level set from OpenVDB file
read bunny.vdb

# Dilate level set:
#   radius of dilation is 5.0 voxels
#   spatial discretization is 5th-order WENO
#   temporal discretization is 1st-order TVD-RK
dilate radius=5 space=5 time=1

# Write level set to file
write bunny_dilated.vdb
