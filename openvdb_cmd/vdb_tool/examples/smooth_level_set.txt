vdb_tool 10.8.0

# This example demonstrates how to read a narrow-band level set from an
# OpenVDB file, smooth it with a Gaussian filter, and write the result
# to a different OpenVDB file.

# Read level set from OpenVDB file
read bunny.vdb

# Gaussian filtering of level set:
#   number of iterations is 5
#   spatial discretization is 5th-order WENO
#   temporal discretization is 1st-order TVD-RK
#   filter-kernel size is 1.0 voxels
gauss iter=5 space=5 time=1 size=1

# Write smoothed level set to file
write bunny_smooth.vdb
